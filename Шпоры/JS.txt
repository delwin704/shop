
/////спойлер с закрытием остальных//////
$(function () {
   $('.services__spoils-title').click(function (event) {
      if ($('.services__spoils').hasClass('one')) {
         $('.services__spoils-title').not($(this)).removeClass('active');
         $('.services__spoils-text').not($(this).next()).slideUp(100);
      }
      $(this).toggleClass('active').next().slideToggle(200); ///// дл€ открыти€ всех
   });
});
